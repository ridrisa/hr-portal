        document.addEventListener('DOMContentLoaded', () => {
            const languageToggle = document.getElementById('languageToggle');
            let isEnglish = true;

            languageToggle.addEventListener('click', () => {
                if (isEnglish) {
                    document.body.style.direction = 'rtl';
                    document.body.style.textAlign = 'right';
                    languageToggle.textContent = 'English';
                    document.querySelector('header h1').textContent = 'مركز المعرفة لموارد البشرية';
                    document.querySelector('h1').textContent = 'فئات مركز المعرفة';
                } else {
                    document.body.style.direction = 'ltr';
                    document.body.style.textAlign = 'left';
                    languageToggle.textContent = 'عربي';
                    document.querySelector('header h1').textContent = 'HR Knowledge Center';
                    document.querySelector('h1').textContent = 'Knowledge Base Categories';
                }
                isEnglish = !isEnglish;
            });

            const searchInput = document.querySelector('.search-bar input');
            const searchButton = document.querySelector('.search-bar button');

            searchButton.addEventListener('click', () => {
                const searchTerm = searchInput.value.toLowerCase();
                const categoryCards = document.querySelectorAll('.category-card');

                categoryCards.forEach(card => {
                    const cardContent = card.textContent.toLowerCase();
                    if (cardContent.includes(searchTerm)) {
                        card.style.display = 'block';
                    } else {
                        card.style.display = 'none';
                    }
                });
            });

            const categoryCards = document.querySelectorAll('.category-card');
            categoryCards.forEach(card => {
                card.addEventListener('mouseenter', () => {
                    card.style.transform = 'translateY(-5px)';
                    card.style.boxShadow = '0 6px 12px rgba(0, 0, 0, 0.15)';
                });
                card.addEventListener('mouseleave', () => {
                    card.style.transform = 'translateY(0)';
                    card.style.boxShadow = '0 4px 6px rgba(0, 0, 0, 0.1)';
                });
            });
        });
    </script>

            <h1>Applicant Tracking Systems (ATS)</h1>

            <p>Applicant Tracking Systems (ATS) are software applications designed to streamline the recruitment process by automating and managing various aspects of talent acquisition. These systems have become essential tools for HR professionals and recruiters, especially in organizations that handle a large volume of job applications.</p>

            <h2>Key Features of ATS</h2>
            <ul>
                <li>Job posting and distribution</li>
                <li>Resume parsing and screening</li>
                <li>Candidate database management</li>
                <li>Application status tracking</li>
                <li>Interview scheduling</li>
                <li>Collaboration tools for hiring teams</li>
                <li>Reporting and analytics</li>
                <li>Integration with HRIS and other HR tools</li>
            </ul>

            <h2>Benefits of Using an ATS</h2>
            <table>
                <tr>
                    <th>Benefit</th>
                    <th>Description</th>
                </tr>
                <tr>
                    <td>Time Savings</td>
                    <td>Automates repetitive tasks, allowing recruiters to focus on high-value activities</td>
                </tr>
                <tr>
                    <td>Improved Candidate Experience</td>
                    <td>Provides a smooth application process and better communication with candidates</td>
                </tr>
                <tr>
                    <td>Enhanced Collaboration</td>
                    <td>Facilitates better communication and decision-making among hiring team members</td>
                </tr>
                <tr>
                    <td>Data-Driven Insights</td>
                    <td>Offers analytics and reporting to optimize recruitment strategies</td>
                </tr>
                <tr>
                    <td>Compliance</td>
                    <td>Helps maintain records and adhere to hiring regulations</td>
                </tr>
            </table>

            <h2>Considerations When Choosing an ATS</h2>
            <ol>
                <li>Scalability: Ensure the system can grow with your organization</li>
                <li>User-friendliness: Look for an intuitive interface for both recruiters and candidates</li>
                <li>Integration capabilities: Check compatibility with your existing HR tech stack</li>
                <li>Customization options: Consider how flexible the system is to meet your specific needs</li>
                <li>Mobile accessibility: Evaluate mobile-friendly features for on-the-go recruitment</li>
                <li>Support and training: Assess the level of customer support and training provided</li>
                <li>Cost: Consider both initial implementation costs and ongoing fees</li>
            </ol>

            <h2>Popular ATS Providers</h2>
            <ul>
                <li>Workday Recruiting</li>

                <li>Lever</li>
                <li>SmartRecruiters</li>
                <li>Jobvite</li>
                <li>BambooHR</li>
            </ul>

            <p>When implementing an ATS, it's crucial to involve key stakeholders, including HR professionals, recruiters, and hiring managers, to ensure the chosen system meets the organization's specific needs and aligns with its recruitment processes.</p>

            <img src="ats-workflow.svg" alt="Diagram showing a typical ATS workflow, including job posting, application review, candidate screening, interview scheduling, and offer management" width="800" height="400">

            <h2>Best Practices for ATS Implementation</h2>
            <ol>
                <li>Define clear objectives and success metrics for your ATS implementation</li>
                <li>Conduct thorough training for all users of the system</li>
                <li>Regularly review and optimize your recruitment workflow within the ATS</li>
                <li>Keep your job descriptions and requirements up-to-date</li>
                <li>Regularly clean and maintain your candidate database</li>
                <li>Use data and analytics to continually improve your recruitment process</li>
                <li>Stay informed about new features and updates from your ATS provider</li>
            </ol>

            <p>By leveraging the power of an Applicant Tracking System, HR professionals can significantly improve their recruitment efficiency, enhance the candidate experience, and make more informed hiring decisions. As the job market becomes increasingly competitive, an effective ATS can be a game-changer for organizations looking to attract and retain top talent.</p>

            <h2>ATS and AI Integration</h2>
            <p>As artificial intelligence (AI) continues to advance, many ATS providers are incorporating AI-driven features to enhance recruitment processes. Some key AI integrations in modern ATS include:</p>
            <ul>
                <li>AI-powered resume screening and ranking</li>
                <li>Chatbots for candidate engagement and initial screening</li>
                <li>Predictive analytics for candidate success and retention</li>
                <li>Automated scheduling and interview coordination</li>
                <li>Natural language processing for job description optimization</li>
            </ul>
            <p>These AI integrations can significantly improve the efficiency and effectiveness of recruitment processes, helping organizations identify and engage top talent more quickly and accurately.</p>

#@title scrap docx
import requests
from bs4 import BeautifulSoup
import os
import time
import logging
from urllib.parse import urljoin
from concurrent.futures import ThreadPoolExecutor, as_completed

# Configure logging
logging.basicConfig(filename='docx_scraper.log', level=logging.INFO,
                    format='%(asctime)s - %(levelname)s - %(message)s')

# Create a console handler for logging to the terminal
console_handler = logging.StreamHandler()
console_handler.setLevel(logging.INFO)  # Log level can be set to DEBUG for more details
formatter = logging.Formatter('%(asctime)s - %(levelname)s - %(message)s')
console_handler.setFormatter(formatter)

# Add the console handler to the root logger
logging.getLogger().addHandler(console_handler)

# Base URL
base_url = "https://hrworld-sa.com"

# Headers to mimic a browser
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/58.0.3029.110 Safari/537.3'
}

def download_docx(url):
    """Downloads a DOCX file from a given URL."""
    try:
        response = requests.get(url, headers=headers, stream=True)
        response.raise_for_status()  # Raise an exception for bad status codes
        filename = os.path.join('docs', os.path.basename(url))
        with open(filename, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                f.write(chunk)
        logging.info(f"Downloaded DOCX: {url} to {filename}")
    except Exception as e:
        logging.error(f"Error downloading {url}: {e}")

def extract_links(url, depth=0, max_depth=7):
    """Extracts DOCX and other links from a webpage."""
    docx_links = set()
    other_links = set()

    if depth > max_depth:
        return docx_links, other_links

    try:
        response = requests.get(url, headers=headers)
        response.raise_for_status()
        soup = BeautifulSoup(response.content, 'html.parser')

        for link in soup.find_all('a', href=True):
            href = link['href']
            href = href.replace('\\', '/')

            # Construct absolute URL
            absolute_url = urljoin(url, href)  # Use urljoin for proper joining

            # Check for DOCX files
            if absolute_url.endswith('.docx') or (link.get('type') and 'docx' in link.get('type')):
                docx_links.add(absolute_url)
                logging.info(f"Found DOCX: {absolute_url}")

            # Check for other links within the base URL (excluding YouTube)
            elif absolute_url.startswith(base_url) and 'youtube.com' not in absolute_url:
                other_links.add(absolute_url)
                logging.debug(f"Found URL: {absolute_url}")

    except Exception as e:
        logging.error(f"Error accessing {url}: {e}")

    return docx_links, other_links

def crawl_website(start_url, max_depth=7):
    """Crawls the website to download DOCX files and extract links."""
    visited_urls = set()
    docx_links, other_links = extract_links(start_url)
    visited_urls.add(start_url)

    # Create directory for downloaded DOCX files
    os.makedirs('docs', exist_ok=True)

    # Download DOCX files
    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = {executor.submit(download_docx, docx_link): docx_link for docx_link in docx_links}

        # Monitor downloads
        for future in as_completed(futures):
            docx_link = futures[future]
            try:
                future.result()  # Wait for the download to complete
            except Exception as e:
                logging.error(f"Error processing {docx_link}: {e}")

    # Recursively crawl other links
    other_links_copy = other_links.copy()
    for other_link in other_links_copy:
        if other_link not in visited_urls:
            visited_urls.add(other_link)
            logging.info(f"Crawling URL: {other_link}")
            new_docx_links, new_other_links = extract_links(other_link, depth=1, max_depth=max_depth)
            docx_links.update(new_docx_links)  # Update original sets
            other_links.update(new_other_links)

            # Download newly found DOCX files
            with ThreadPoolExecutor(max_workers=5) as executor:
                futures = {executor.submit(download_docx, docx_link): docx_link for docx_link in new_docx_links}

                for future in as_completed(futures):
                    docx_link = futures[future]
                    try:
                        future.result()  # Wait for the download to complete
                    except Exception as e:
                        logging.error(f"Error processing {docx_link}: {e}")

if __name__ == "__main__":
    crawl_website(base_url)

#@title scrap pdfs
import requests
from bs4 import BeautifulSoup
import os
import time
import logging
from urllib.parse import urljoin
from concurrent.futures import ThreadPoolExecutor, as_completed

# Configure logging
logging.basicConfig(filename='pdf_scraper.log', level=logging.INFO,
                    format='%(asctime)s - %(levelname)s - %(message)s')
# Create a console handler for logging to the terminal
console_handler = logging.StreamHandler()
console_handler.setLevel(logging.INFO)  # Log level can be set to DEBUG for more details
formatter = logging.Formatter('%(asctime)s - %(levelname)s - %(message)s')
console_handler.setFormatter(formatter)

# Add the console handler to the root logger
logging.getLogger().addHandler(console_handler)

# Base URL
base_url = "https://hrworld-sa.com"

# Headers to mimic a browser
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/58.0.3029.110 Safari/537.3'
}

def download_pdf(url):
    """Downloads a PDF file from a given URL."""
    try:
        response = requests.get(url, headers=headers, stream=True)
        response.raise_for_status()  # Raise an exception for bad status codes
        filename = os.path.join('pdfs', os.path.basename(url))
        with open(filename, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                f.write(chunk)
        logging.info(f"Downloaded PDF: {url} to {filename}")
    except Exception as e:
        logging.error(f"Error downloading {url}: {e}")

def extract_links(url, depth=0, max_depth=7):
    """Extracts PDF and other links from a webpage."""
    pdf_links = set()
    other_links = set()

    if depth > max_depth:
        return pdf_links, other_links

    try:
        response = requests.get(url, headers=headers)
        response.raise_for_status()
        soup = BeautifulSoup(response.content, 'html.parser')

        for link in soup.find_all('a', href=True):
            href = link['href']
            href = href.replace('\\', '/')

            # Construct absolute URL
            absolute_url = urljoin(url, href)  # Use urljoin for proper joining

            # Check for PDF files
            if absolute_url.endswith('.pdf') or (link.get('type') and 'pdf' in link.get('type')):
                pdf_links.add(absolute_url)
                logging.info(f"Found PDF: {absolute_url}")

            # Check for other links within the base URL (excluding YouTube)
            elif absolute_url.startswith(base_url) and 'youtube.com' not in absolute_url:
                other_links.add(absolute_url)
                logging.debug(f"Found URL: {absolute_url}")

    except Exception as e:
        logging.error(f"Error accessing {url}: {e}")

    return pdf_links, other_links

def crawl_website(start_url, max_depth=7):
    """Crawls the website to download PDFs and extract links."""
    visited_urls = set()
    pdf_links, other_links = extract_links(start_url)
    visited_urls.add(start_url)

    # Create directory for downloaded PDFs
    os.makedirs('pdfs', exist_ok=True)

    # Download PDFs
    with ThreadPoolExecutor(max_workers=5) as executor:
        futures = {executor.submit(download_pdf, pdf_link): pdf_link for pdf_link in pdf_links}

        # Monitor downloads
        for future in as_completed(futures):
            pdf_link = futures[future]
            try:
                future.result()  # Wait for the download to complete
            except Exception as e:
                logging.error(f"Error processing {pdf_link}: {e}")

    # Recursively crawl other links
    other_links_copy = other_links.copy()
    for other_link in other_links_copy:
        if other_link not in visited_urls:
            visited_urls.add(other_link)
            logging.info(f"Crawling URL: {other_link}")
            new_pdf_links, new_other_links = extract_links(other_link, depth=1, max_depth=max_depth)
            pdf_links.update(new_pdf_links)  # Update original sets
            other_links.update(new_other_links)

            # Download newly found PDFs
            with ThreadPoolExecutor(max_workers=5) as executor:
                futures = {executor.submit(download_pdf, pdf_link): pdf_link for pdf_link in new_pdf_links}

                for future in as_completed(futures):
                    pdf_link = futures[future]
                    try:
                        future.result()  # Wait for the download to complete
                    except Exception as e:
                        logging.error(f"Error processing {pdf_link}: {e}")

if __name__ == "__main__":
    crawl_website(base_url)

        function calculateSalaryRange() {
            const midpoint = parseFloat(document.getElementById('midpoint').value);
            const spread = parseFloat(document.getElementById('spread').value) / 100;

            if (isNaN(midpoint) || isNaN(spread)) {
                document.getElementById('result').innerHTML = 'Please enter valid numbers for both fields.';
                return;
            }

            const minimum = midpoint / (1 + (spread / 2));
            const maximum = midpoint * (1 + (spread / 2));

            document.getElementById('result').innerHTML = `
                Salary Range:<br>
                Minimum: $${minimum.toFixed(2)}<br>
                Midpoint: $${midpoint.toFixed(2)}<br>
                Maximum: $${maximum.toFixed(2)}
            `;
        }

            // Animate counter function
            function animateCounter(elementId, targetValue, duration) {
                let current = 0;
                const element = document.getElementById(elementId);
                const increment = targetValue / (duration / 16); // 16ms per frame (60fps)
                const timer = setInterval(() => {
                    current += increment;
                    element.textContent = Math.floor(current);
                    if (current >= targetValue) {
                        clearInterval(timer);
                        element.textContent = targetValue;
                    }
                }, 16);
            }
        
            // Set up Intersection Observer for counter animation
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        animateCounter('members-count', 5000, 2000); // Example target values
                        animateCounter('resources-count', 1000, 2000);
                        animateCounter('events-count', 50, 2000);
                        animateCounter('jobs-count', 500, 2000);
                        observer.unobserve(entry.target); // Stop observing after animation
                    }
                });
            }, { threshold: 0.5 });
        
            observer.observe(document.getElementById('stats')); // Observe the stats section
        </script>

        <h1>Pay for Performance</h1>

        <p>Pay for Performance (PFP) is a compensation strategy that links an employee's pay to their performance or productivity. This approach aims to motivate employees, increase productivity, and align individual goals with organizational objectives. In the Kingdom of Saudi Arabia (KSA), as businesses strive to enhance efficiency and competitiveness in line with Vision 2030, PFP systems are gaining popularity. The shift towards a more diversified economy and the emphasis on private sector growth have made effective performance management crucial for organizations operating in the Kingdom.</p>

        <h2>Key Components of Pay for Performance</h2>

        <ul>
            <li>Clear performance metrics and goals</li>
            <li>Regular performance evaluations</li>
            <li>Transparent reward systems</li>
            <li>Alignment with organizational objectives</li>
            <li>Flexible compensation structures</li>
        </ul>

        <h2>Types of Pay for Performance Systems</h2>

        <p>There are several types of Pay for Performance systems that organizations in KSA can consider:</p>

        <ol>
            <li><strong>Merit-based Pay:</strong> Annual salary increases based on performance evaluations.</li>
            <li><strong>Bonus Plans:</strong> One-time payments for achieving specific goals or targets.</li>
            <li><strong>Profit-sharing:</strong> Distributing a portion of company profits to employees.</li>
            <li><strong>Gain-sharing:</strong> Rewarding employees for improvements in productivity or cost savings.</li>
            <li><strong>Commission:</strong> Common in sales roles, where pay is directly tied to sales performance.</li>
            <li><strong>Skill-based Pay:</strong> Compensating employees for developing specific skills or competencies.</li>
        </ol>

        <p>Each system has its advantages and may be more suitable for different industries or organizational structures within the Saudi context.</p>

        <h2>Benefits of Pay for Performance</h2>

        <ol>
            <li><strong>Increased motivation:</strong> Employees are incentivized to perform at their best.</li>
            <li><strong>Improved productivity:</strong> Clear goals and rewards can lead to higher output.</li>
            <li><strong>Retention of top talent:</strong> High performers are rewarded, encouraging them to stay with the company.</li>
            <li><strong>Alignment with company goals:</strong> Individual performance is tied to organizational objectives.</li>
            <li><strong>Fairness in compensation:</strong> Rewards are based on measurable performance rather than subjective factors.</li>
        </ol>

        <div class="info-box">
            <h3>KSA Context</h3>
            <p>In Saudi Arabia, implementing PFP systems should consider cultural nuances, such as the emphasis on collective achievement and the importance of maintaining harmonious workplace relationships. It's crucial to design a system that balances individual recognition with team cohesion.</p>
        </div>

        <h2>Implementing Pay for Performance in KSA</h2>

        <h3>1. Establish Clear Metrics</h3>
        <p>Define specific, measurable, achievable, relevant, and time-bound (SMART) goals for each role. Ensure these metrics align with the company's overall objectives and consider both quantitative and qualitative measures.</p>

        <h3>2. Communicate Effectively</h3>
        <p>Clearly explain the PFP system to all employees. Transparency is key to gaining buy-in and ensuring fairness. Regular communication about performance expectations and progress is essential.</p>

        <h3>3. Provide Regular Feedback</h3>
        <p>Implement a system for ongoing performance discussions. This could include monthly check-ins, quarterly reviews, and annual evaluations. Timely feedback helps employees stay on track and make necessary adjustments.</p>

        <h3>4. Offer Various Incentives</h3>
        <p>Consider a mix of short-term and long-term incentives. This could include bonuses, salary increases, profit-sharing, or stock options. Tailor the rewards to what motivates your workforce.</p>

        <h3>5. Ensure Fairness and Consistency</h3>
        <p>Implement checks and balances to prevent bias in performance evaluations. Consider using calibration sessions where managers discuss and align on performance ratings across teams.</p>

        <h3>6. Adapt to KSA Labor Laws</h3>
        <p>Ensure your PFP system complies with Saudi labor laws, particularly regarding minimum wage requirements and any regulations on variable pay components.</p>

        <h3>7. Align with Islamic Principles</h3>
        <p>Ensure that the PFP system aligns with Islamic principles of fairness and ethical business practices. This can include avoiding excessive risk or speculation in bonus structures and ensuring that performance metrics do not encourage unethical behavior.</p>

        <h3>8. Consider Saudization Goals</h3>
        <p>Design your PFP system to support Saudization efforts by including metrics that reward the development and promotion of Saudi nationals within the organization.</p>

        <h3>9. Integrate with Existing HR Systems</h3>
        <p>Ensure that your PFP system integrates seamlessly with existing HR management systems, including those that comply with local regulations such as the General Organization for Social Insurance (GOSI) requirements.</p>

        <h2>Challenges and Considerations in KSA</h2>

        <p>Implementing Pay for Performance systems in Saudi Arabia comes with unique challenges:</p>

        <ul>
            <li><strong>Cultural Sensitivity:</strong> Balance individual recognition with the collective-oriented culture prevalent in KSA.</li>
            <li><strong>Gender Considerations:</strong> Ensure that PFP systems are equitable across genders, particularly as more women enter the workforce under Vision 2030 initiatives.</li>
            <li><strong>Expatriate vs. Local Workforce:</strong> Design systems that are fair to both expatriate and Saudi employees while supporting Saudization goals.</li>
            <li><strong>Regulatory Compliance:</strong> Stay updated with rapidly evolving labor laws and ensure PFP systems comply with all local regulations.</li>
            <li><strong>Performance Measurement in New Industries:</strong> Develop appropriate metrics for emerging industries that are part of KSA's economic diversification efforts.</li>
        </ul>

        <h2>Conclusion</h2>
        <p>Pay for Performance can be a powerful tool for driving employee engagement and organizational success in Saudi Arabia. When implemented thoughtfully, it can support the Kingdom's economic transformation goals while motivating the workforce. However, it requires careful planning, clear communication, and ongoing management to be effective. By tailoring the PFP approach to the unique cultural and business environment of KSA, companies can create a motivating and fair compensation system that benefits both employees and the organization.</p>

        <p>Looking ahead, the future of PFP in KSA is likely to evolve alongside the country's economic development. We may see more sophisticated systems emerge that incorporate AI and data analytics for performance evaluation, as well as a greater emphasis on long-term incentives aligned with the country's strategic goals. As the workforce becomes more diverse and specialized, PFP systems will need to remain flexible and adaptive to meet the changing needs of both employees and organizations in the Kingdom.</p>

        function calculateBenefits() {
            const salary = parseFloat(document.getElementById('salary').value);
            const healthInsurance = parseFloat(document.getElementById('health-insurance').value);
            const retirementMatch = parseFloat(document.getElementById('retirement-match').value);

            if (isNaN(salary) || isNaN(healthInsurance) || isNaN(retirementMatch)) {
                alert('Please enter valid numbers for all fields.');
                return;
            }

            const annualHealthInsurance = healthInsurance * 12;
            const annualRetirementMatch = salary * (retirementMatch / 100);
            const totalValue = salary + annualHealthInsurance + annualRetirementMatch;

            const resultDiv = document.getElementById('result');
            resultDiv.innerHTML = `
                <p>Estimated Total Value of Compensation Package:</p>
                <p>$${totalValue.toFixed(2)}</p>
                <p>This includes:</p>
                <ul>
                    <li>Salary: $${salary.toFixed(2)}</li>
                    <li>Annual Health Insurance Value: $${annualHealthInsurance.toFixed(2)}</li>
                    <li>Annual 401(k) Match: $${annualRetirementMatch.toFixed(2)}</li>
                </ul>
            `;
        }
    </script>

        <h1>Mastering the Art of Interviewing</h1>

        <p>Effective interviewing is crucial for identifying the best candidates for your organization. This guide covers various interviewing techniques to help you make informed hiring decisions.</p>

        <div class="technique">
            <h2>1. Behavioral Interviewing</h2>
            <p>This technique focuses on past experiences to predict future performance.</p>
            <div class="example">
                <strong>Example question:</strong> "Tell me about a time when you had to deal with a difficult team member. How did you handle the situation?"
            </div>
            <div class="tip">
                <strong>Tip:</strong> Use the STAR method (Situation, Task, Action, Result) to evaluate responses.
            </div>
        </div>

        <div class="technique">
            <h2>2. Situational Interviewing</h2>
            <p>Present hypothetical scenarios to assess a candidate's problem-solving skills.</p>
            <div class="example">
                <strong>Example question:</strong> "If you were assigned a project with an impossible deadline, how would you approach it?"
            </div>
            <div class="tip">
                <strong>Tip:</strong> Look for candidates who demonstrate creativity, prioritization, and communication skills in their responses.
            </div>
        </div>

        <div class="technique">
            <h2>3. Competency-Based Interviewing</h2>
            <p>Focus on specific skills and competencies required for the role.</p>
            <div class="example">
                <strong>Example question:</strong> "Describe a project where you demonstrated strong leadership skills. What was the outcome?"
            </div>
            <div class="tip">
                <strong>Tip:</strong> Align your questions with the key competencies outlined in the job description.
            </div>
        </div>

        <div class="technique">
            <h2>4. Structured Interviewing</h2>
            <p>Use a standardized set of questions for all candidates to ensure fairness and comparability.</p>
            <div class="example">
                <strong>Example approach:</strong> Create a scoring rubric for each question and use it consistently across all interviews.
            </div>
            <div class="tip">
                <strong>Tip:</strong> Train all interviewers on the structured process to maintain consistency.
            </div>

                    <li>Identify the root cause of the conflict</li>
                    <li>Brainstorm possible solutions</li>
                    <li>Agree on a mutually beneficial resolution</li>

        <h1>Writing Effective Job Descriptions</h1>

        <p>A well-written job description is crucial for attracting the right candidates and setting clear expectations for the role. This guide will help you create compelling and effective job descriptions.</p>

        <div class="highlight-box">
            <h3>Key Elements of an Effective Job Description:</h3>
            <ul class="checklist">
                <li>Clear and concise job title</li>
                <li>Brief company overview</li>
                <li>Job summary</li>
                <li>Key responsibilities</li>
                <li>Required qualifications and skills</li>
                <li>Preferred qualifications</li>
                <li>Information about the work environment</li>
                <li>Salary range and benefits (if applicable)</li>
                <li>Equal opportunity statement</li>
            </ul>
        </div>

        <h2>1. Job Title</h2>
        <p>Use a clear, specific title that accurately reflects the role. Avoid internal jargon or overly creative titles that might confuse job seekers.</p>

        <div class="example">
            <strong>Good Example:</strong> "Senior Software Engineer - Java"<br>
            <strong>Poor Example:</strong> "Code Ninja"
        </div>

        <h2>2. Company Overview</h2>
        <p>Provide a brief, engaging description of your company, its mission, and culture. This helps candidates understand if they'd be a good fit.</p>

        <h2>3. Job Summary</h2>
        <p>Offer a concise overview of the role, its place within the organization, and its main objectives.</p>

        <h2>4. Key Responsibilities</h2>
        <p>List the primary duties and responsibilities of the position. Use action verbs and be specific about what the role entails.</p>

        <div class="example">
            <strong>Example:</strong>
            <ul>
                <li>Develop and maintain Java-based applications</li>
                <li>Collaborate with cross-functional teams to define and implement new features</li>
                <li>Participate in code reviews and contribute to best practices</li>
            </ul>
        </div>

        <h2>5. Required Qualifications and Skills</h2>
        <p>Clearly state the minimum qualifications needed for the role, including education, experience, and specific skills.</p>

        <h2>6. Preferred Qualifications</h2>
        <p>List any additional qualifications that would be beneficial but are not required.</p>

        <h2>7. Work Environment</h2>
        <p>Describe the work setting, team structure, and any unique aspects of the role (e.g., travel requirements, remote work options).</p>

        <h2>8. Salary and Benefits</h2>
        <p>If possible, include a salary range and highlight key benefits to attract top talent.</p>

        <h2>9. Equal Opportunity Statement</h2>
        <p>Include a statement affirming your commitment to equal opportunity employment and diversity.</p>

import os

def list_directory_structure(directory_path, output_file, indent=""):
    for item in os.listdir(directory_path):
        item_path = os.path.join(directory_path, item)
        if os.path.isdir(item_path):
            output_file.write(f"{indent}📂 {item}/\n")
            list_directory_structure(item_path, output_file, indent + "    ")
        else:
            output_file.write(f"{indent}📄 {item}\n")

# Get the current working directory
current_directory = os.getcwd()

# Define the output file path
output_file_path = "directory_structure.txt"

# Open the file and write the directory structure to it
with open(output_file_path, "w") as file:
    list_directory_structure(current_directory, file)

import os
from PyPDF2 import PdfMerger

# Create a PdfMerger object
merger = PdfMerger()

# List all PDF files in the current directory
pdf_files = [f for f in os.listdir('.') if f.endswith('.pdf')]

# Check if there are PDF files to merge
if not pdf_files:
    print("No PDF files found in the current directory.")
else:
    # Add each PDF file to the merger
    for pdf in pdf_files:
        merger.append(pdf)
        print(f"Added: {pdf}")

    # Specify the name for the merged PDF
    merged_pdf_name = "merged.pdf"
    
    # Write out the merged PDF
    merger.write(merged_pdf_name)
    merger.close()

    print(f"Merged {len(pdf_files)} PDF files into {merged_pdf_name}.")
